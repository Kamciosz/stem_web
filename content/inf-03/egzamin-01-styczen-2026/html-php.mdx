## Srodkowa sekcja — dwa losowe + rowspan

Dwa losowe pojazdy z ORDER BY RAND() LIMIT 2. Kazdy wyswietlany z obrazkiem (a1.jpg, a2.jpg)
i tabelka z rowspan="2".

<ExamCodeGrid>
  <ExamCodeBlock
    language="PHP"
    title="Losowanie i rowspan"
    description="Dwa auta, dwa obrazki, tabela z rowspan='2' na kolumnie ceny."
    mistake="Brak rowspan — uklad wizualnie podobny do makiety, ale wymaganie niespelnione."
    code={$query = "SELECT marka, model, cena FROM pojazdy ORDER BY RAND() LIMIT 2";
$result = $conn->query($query);
$nr = 1;

while ($row = $result->fetch_assoc()) {
    echo "<img src='a{$nr}.jpg' alt='Pojazd {$nr}'>";
    echo "<table>";
    echo "<tr><td rowspan='2'>" . $row['cena'] . "</td>";
    echo "<td>Marka: " . $row['marka'] . "</td></tr>";
    echo "<tr><td>Model: " . $row['model'] . "</td></tr>";
    echo "</table>";
    $nr++;
}}
  />
</ExamCodeGrid>
